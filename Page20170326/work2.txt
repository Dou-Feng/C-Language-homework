#include "stdio.h"

void main() {
    float n1, n2;
    char c;
    int t;
    char temp = 'q';
    while (temp != EOF) {
        scanf("%d%f%f%c%c", &t, &n1, &n2, &c, &temp);
        switch (t) {
            case 1:
                switch (c) {
                    case '+':
                        printf("After switch processing,the result is : %.2f\n", n1 + n2);
                        break;
                    case '-':
                        printf("After switch processing,the result is : %.2f\n", n1 - n2);
                        break;
                    case '*':
                        printf("After switch processing,the result is : %.2f\n", n1 * n2);
                        break;
                    case '/':
                        printf("After switch processing,the result is : %.2f\n", n1 / n2);
                        break;
                }
                break;
            default:
                break;
        }
        if (t == 0) {
            if (c == '+') {
                printf("After switch processing,the result is : %.2f\n", n1 + n2);
            } else if (c == '-') {
                printf("After switch processing,the result is : %.2f\n", n1 - n2);
            } else if (c == '*') {
                printf("After switch processing,the result is : %.2f\n", n1 * n2);
            } else if (c == '/') {
                printf("After switch processing,the result is : %.2f\n", n1 / n2);
            }
        }
        if (t == 2) {
            if (c == '+') {
                printf("After if-else processing,the result is : %.2f\n", n1 + n2);
                printf("After switch processing,the result is : %.2f\n", n1 + n2);
            } else if (c == '-') {
                printf("After if-else processing,the result is : %.2f\n", n1 - n2);
                printf("After switch processing,the result is : %.2f\n", n1 - n2);
            } else if (c == '*') {
                printf("After if-else processing,the result is : %.2f\n", n1 * n2);
                printf("After switch processing,the result is : %.2f\n", n1 * n2);
            } else if (c == '/') {
                printf("After if-else processing,the result is : %.2f\n", n1 / n2);
                printf("After switch processing,the result is : %.2f\n", n1 / n2);
            }
        }
        printf("\n");
    }
}
